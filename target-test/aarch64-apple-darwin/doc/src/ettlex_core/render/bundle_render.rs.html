5/// Render a leaf bundle to Markdown
6///
7/// Generates a complete Markdown bundle by computing the EPT from root to leaf,
8/// then aggregating the WHY/WHAT/HOW content from all EPs in the path.
9///
10/// The output includes:
11/// - Title showing the full RT path (Root > Mid > Leaf)
12/// - Aggregated WHY sections from all EPs
13/// - Aggregated WHAT sections from all EPs
14/// - Aggregated HOW sections from all EPs
15///
16/// # Arguments
17/// * `store` - Reference to the Store
18/// * `leaf_id` - ID of the leaf Ettle
19/// * `leaf_ep_ordinal` - Optional ordinal of the leaf EP to end with
20///
21/// # Returns
22/// Markdown string representation of the bundle
23///
24/// # Errors
25/// * `EttleNotFound` - If leaf doesn't exist
26/// * `EptMissingMapping` / `EptDuplicateMapping` - EPT computation errors
27/// * `EptAmbiguousLeafEp` - If leaf has multiple EPs and ordinal not specified
28pub fn render_leaf_bundle(
29    store: &Store,
30    leaf_id: &str,
31    leaf_ep_ordinal: Option<u32>,
32) -> Result<String> {
33    // Compute EPT
34    let ept = compute_ept(store, leaf_id, leaf_ep_ordinal)?;
35
36    let mut output = String::new();
37
38    // Compute RT for title path
39    let rt = crate::traversal::rt::compute_rt(store, leaf_id)?;
40    let path: Vec<String> = rt
41        .iter()
42        .filter_map(|ettle_id| store.get_ettle(ettle_id).ok().map(|e| e.title.clone()))
43        .collect();
44
45    // Title
46    output.push_str(&format!("# Leaf Bundle: {}\n\n", path.join(" > ")));
47
48    // Aggregate WHY sections
49    let mut why_sections = Vec::new();
50    for ep_id in &ept {
51        if let Ok(ep) = store.get_ep(ep_id) {
52            if !ep.why.is_empty() {
53                why_sections.push(ep.why.clone());
54            }
55        }
56    }
57
58    if !why_sections.is_empty() {
59        output.push_str("## WHY (Rationale)\n\n");
60        for (i, why) in why_sections.iter().enumerate() {
61            output.push_str(&format!("{}. {}\n", i + 1, why));
62        }
63        output.push('\n');
64    }
65
66    // Aggregate WHAT sections
67    let mut what_sections = Vec::new();
68    for ep_id in &ept {
69        if let Ok(ep) = store.get_ep(ep_id) {
70            if !ep.what.is_empty() {
71                what_sections.push(ep.what.clone());
72            }
73        }
74    }
75
76    if !what_sections.is_empty() {
77        output.push_str("## WHAT (Description)\n\n");
78        for (i, what) in what_sections.iter().enumerate() {
79            output.push_str(&format!("{}. {}\n", i + 1, what));
80        }
81        output.push('\n');
82    }
83
84    // Aggregate HOW sections
85    let mut how_sections = Vec::new();
86    for ep_id in &ept {
87        if let Ok(ep) = store.get_ep(ep_id) {
88            if !ep.how.is_empty() {
89                how_sections.push(ep.how.clone());
90            }
91        }
92    }
93
94    if !how_sections.is_empty() {
95        output.push_str("## HOW (Implementation)\n\n");
96        for (i, how) in how_sections.iter().enumerate() {
97            output.push_str(&format!("{}. {}\n", i + 1, how));
98        }
99        output.push('\n');
100    }
101
102    Ok(output)
103}

105#[cfg(test)]
106mod tests {
107    use super::*;
108    use crate::model::{Ep, Ettle};
109
110    #[test]
111    fn test_render_leaf_bundle_single_ettle() {
112        let mut store = Store::new();
113
114        let mut ettle = Ettle::new("root".to_string(), "Root".to_string());
115        let ep0 = Ep::new(
116            "ep0".to_string(),
117            "root".to_string(),
118            0,
119            true,
120            "Why 0".to_string(),
121            "What 0".to_string(),
122            "How 0".to_string(),
123        );
124
125        ettle.add_ep_id("ep0".to_string());
126        store.insert_ettle(ettle);
127        store.insert_ep(ep0);
128
129        let output = render_leaf_bundle(&store, "root", None).unwrap();
130
131        assert!(output.contains("# Leaf Bundle: Root"));
132        assert!(output.contains("Why 0"));
133        assert!(output.contains("What 0"));
134        assert!(output.contains("How 0"));
135    }
